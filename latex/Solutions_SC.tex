\end{itemize}

\section{QCM théoriques et questions de cours}
\markboth{QCM théoriques}{}

\subsection*{Concepts de base et décomposition}

\textbf{Q\,A.1.} \begin{reponse}\bonne{A}.\end{reponse}
Un processus SSL a $\E(X_t)=m$ \emph{constante} et $\gamma(k)$ indépendante de $t$. Une
tendance rend $\E(X_t)$ fonction de $t$ ; une saisonnalité crée une dépendance périodique en
$t$. Dans les deux cas la stationnarité est violée : un modèle stationnaire ne peut donc
décrire \emph{ni} tendance \emph{ni} saisonnalité (on stationnarise d'abord).

\textbf{Q\,A.2.} \begin{reponse}\bonne{B}.\end{reponse}
Définition : MM $=$ filtre \emph{linéaire} $\sum_i\lambda_iX_{t-i}$. Être centré ($m_1=m_2$),
symétrique ($\lambda_{-i}=\lambda_i$) ou normalisé ($\sum\lambda_i=1$) sont des propriétés
\emph{en plus}, non requises par la définition.

\textbf{Q\,A.3.} \begin{reponse}\bonne{C} (conserve les tendances linéaires, absorbe la
saisonnalité de période 4).\end{reponse}
Théorème du cours : une MM centrée d'ordre $p$ laisse invariante une tendance affine et annule
une saisonnalité de période $p$. Avec $p=4$ : on garde la tendance, on supprime la composante
trimestrielle.

\textbf{Q\,A.4.} \begin{reponse}Correctes : \bonne{A} et \bonne{C}.\end{reponse}
X11 estime la CVS de façon \emph{itérative} (boucles) en combinant régression et moyennes
mobiles. « Le plus sophistiqué connu » est faux (X-12-ARIMA, X-13ARIMA-SEATS lui succèdent).

\textbf{Q\,A.5.} \begin{reponse}\bonne{Faux}.\end{reponse}
La décomposition vise à \emph{isoler les trois composantes} $T_t,S_t,E_t$. Retirer la seule
saisonnalité (désaisonnalisation/CVS) n'en est qu'une application.

\textbf{Q\,A.6.} \begin{reponse}\bonne{Vrai}.\end{reponse}
Une bonne première MM doit \emph{conserver} la tendance, \emph{absorber} la saisonnalité et
\emph{atténuer} la variance du bruit ($\Var/p$) : c'est le rôle de la MM d'ordre $p$.

\textbf{Q\,A.7.} \begin{reponse}\bonne{Vrai}.\end{reponse}
Le lissage exponentiel simple ajuste localement une \emph{constante} : aucune composante de
tendance ni de saisonnalité.

\textbf{Q\,A.8.} \begin{reponse}Correctes : \bonne{A} et \bonne{C}.\end{reponse}
Holt--Winters généralise les lissages simple/double et intègre \emph{tendance ET saisonnalité}
(item B faux : il prend bien en compte la tendance).

\subsection*{Caractérisation AR / MA / ARMA / SARIMA}

\textbf{Q\,A.9.} \begin{reponse}\bonne{B}.\end{reponse}
AR$(p)$ : ACF \emph{décroît} (exponentielle/sinus amortie), PACF \emph{s'annule} pour $k>p$.

\textbf{Q\,A.10.} \begin{reponse}\bonne{D}.\end{reponse}
ARMA : \emph{ni} l'ACF \emph{ni} la PACF ne coupent ; les deux décroissent (mélange AR+MA).

\textbf{Q\,A.11.} \begin{reponse}\bonne{D} (tendance et/ou saisonnalité).\end{reponse}
SARIMA $=$ ARIMA $\times$ partie saisonnière : $(1-B)^d$ gère la tendance, $(1-B^s)^D$ la
saisonnalité. Il n'est pas stationnaire en général (A faux).

\textbf{Q\,A.12.} \begin{reponse}\bonne{Faux}.\end{reponse}
Un MA est toujours stationnaire mais n'a une représentation AR que s'il est \emph{inversible}.
Contre-exemple : $X_t=a_t-a_{t-1}$ (racine $B=1$).

\textbf{Q\,A.13.} \begin{reponse}Essentiel : \bonne{B} ; en pratique \bonne{A} et \bonne{C}
aussi.\end{reponse}
Le c\oe ur est la \emph{blancheur des résidus} (Ljung--Box). On contrôle en complément la
significativité (Student) et la normalité (Jarque--Bera, pour les IC).

\textbf{Q\,A.14.} \begin{reponse}\bonne{Faux}.\end{reponse}
Stationnaire $\ne$ bruit blanc : un AR(1) stationnaire a $\rho(k)=\phi^k\ne0$, donc Ljung--Box
\emph{rejette} $H_0$. Ce test s'applique aux \emph{résidus} d'un modèle, pas à la série brute.

\subsection*{Questions de cours}

\textbf{Q\,A.15.} \emph{Strict} (loi invariante par translation) vs \emph{SSL} ($\E(X_t)$
constante, $\gamma(k)$ indépendante de $t$). La SSL est plus \emph{pratique} : elle ne porte
que sur les deux premiers moments, estimables empiriquement, alors que la stationnarité stricte
porte sur toute la loi jointe.

\textbf{Q\,A.16.} Si $(X_t)$ est strictement stationnaire \emph{et} $\E(X_t^2)<\infty$, alors
il est SSL : l'invariance de la loi entraîne celle de $\E(X_t)$ et de $\gamma(k)$ (qui existent
grâce au carré intégrable).

\textbf{Q\,A.17.} Le corrélogramme (ACF empirique) identifie : la \emph{non-stationnarité}
(décroissance lente $\approx1$), la \emph{saisonnalité} (pics aux multiples de la période),
l'\emph{ordre $q$ d'un MA} (coupure après $q$).

\textbf{Q\,A.18.} Un AR$(p)$ est caractérisé par une FAP qui \emph{s'annule pour $k>p$}
($P_p=\phi_p\ne0$) : la PACF « coupe » à l'ordre $p$.

\textbf{Q\,A.19.} (i) \emph{Ljung--Box} sur les résidus :
$\mathrm{LB}=n(n+2)\sum_{k=1}^K\frac{\rhohat_k^2}{n-k}\sim\chi^2_{K-p-q}$, $H_0$ : bruit blanc.
(ii) \emph{Jarque--Bera} (normalité) ou test de Student (significativité des paramètres).

\textbf{Q\,A.20.} Critères d'erreur hors échantillon : RMSE, MAE, MAPE (ou $R^2$ ajusté). Ils
mesurent l'écart entre $X_{n+i}$ et $\widehat X_{n+i}$.

\textbf{Q\,A.21.} \emph{AIC}$=-2\ln L+2k$ et \emph{BIC}$=-2\ln L+k\ln n$ : on \emph{minimise} ;
le BIC pénalise plus la complexité (modèles plus parcimonieux).

\section{Stationnarité, causalité et inversibilité}
\markboth{Stationnarité, causalité, inversibilité}{}

\textbf{Q\,B.1 --- $X_t=(-1)^tz$.}
\begin{etapes}
$\E(X_t)=(-1)^t\E(z)=0$. Puis
$\gamma(k)=\Cov(X_t,X_{t+k})=\E[X_tX_{t+k}]=(-1)^t(-1)^{t+k}\E(z^2)=(-1)^{2t+k}\cdot1=(-1)^k$.
Comme $2t$ est pair, $(-1)^{2t+k}=(-1)^k$ : indépendant de $t$.
\end{etapes}
\begin{reponse}$\E(X_t)=0$, $\gamma(k)=(-1)^k$ (donc $\gamma(0)=1$, $\rho(k)=(-1)^k$) :
\bonne{$(X_t)$ est stationnaire au sens large}.\end{reponse}
\emph{Remarque.} Le processus n'est pas ergodique (toute la trajectoire dépend du seul tirage
$z$), mais la SSL ne concerne que les deux premiers moments.

\textbf{Q\,B.2 --- $X_t=0.8X_{t-1}-0.15X_{t-2}+a_t-0.3a_{t-1}$.}
$\Phi(B)=1-0.8B+0.15B^2$, $\Theta(B)=1-0.3B$.
\begin{etapes}
\emph{Stationnarité} : racines de $0.15B^2-0.8B+1=0$. Discriminant $0.64-0.6=0.04$,
$B=\frac{0.8\pm0.2}{0.3}\in\{2,\ 3.33\}$ : modules $>1\Rightarrow$ stationnaire.
\emph{Inversibilité} : $1-0.3B=0\Rightarrow B=3.33>1\Rightarrow$ inversible.
\end{etapes}
\begin{reponse}\bonne{(a) stationnaire et inversible}.\end{reponse}

\textbf{Q\,B.3 --- $(1-1.66B+0.66B^2)X_t=(1-B)a_t$.}
\begin{etapes}
Factorisons : $1-1.66B+0.66B^2=(1-B)(1-0.66B)$ (vérif.\ : $(1-B)(1-0.66B)=1-1.66B+0.66B^2$).
L'équation devient $(1-B)(1-0.66B)X_t=(1-B)a_t$. Le facteur \emph{commun} $(1-B)$ se simplifie :
\[
(1-0.66B)X_t=a_t,\quad\text{soit}\quad X_t=0.66X_{t-1}+a_t.
\]
\end{etapes}
\begin{reponse}\bonne{Vrai} : après simplification c'est un AR(1) ($0.66<1$, stationnaire),
qui est une représentation \emph{AR d'ordre fini}.\end{reponse}
\begin{attention}Sans repérer la racine commune $B=1$, on conclurait faussement à la
non-inversibilité.\end{attention}

\textbf{Q\,B.4 --- $X_t=1.6X_{t-1}-0.6X_{t-2}+a_t-0.6a_{t-1}$.}
\begin{etapes}
$\Phi(B)=1-1.6B+0.6B^2=(1-B)(1-0.6B)$ : racine $B=1$ (module $=1$, pas $>1$). De plus
$\Theta(B)=1-0.6B$ se simplifie avec le facteur $(1-0.6B)$ de $\Phi$, laissant
$(1-B)X_t=a_t$ : \emph{marche aléatoire}.
\end{etapes}
\begin{reponse}\bonne{Faux} : la racine unité $B=1$ rend $X_t$ non stationnaire (intégré
$I(1)$).\end{reponse}

\textbf{Q\,B.5 --- $(1-0.8B)X_t=(1+0.4B)a_t$.}
$\phi_1=0.8$ : racine $B=1.25>1$ (stationnaire/causal). $\Theta(B)=1+0.4B$ : racine
$B=-2.5$, $|{-2.5}|>1$ (inversible).
\begin{reponse}
\emph{(a)} MA inversible $\Rightarrow$ on inverse $\Theta$ : \bonne{représentation
$\mathrm{AR}(\infty)$ (ordre infini)}.
\emph{(b)} AR causal $\Rightarrow$ on inverse $\Phi$ : \bonne{représentation
$\mathrm{MA}(\infty)$ (ordre infini)}. (Coefficients : Q\,C.3.)
\end{reponse}

\textbf{Q\,B.6 --- $X_t=0.8X_{t-1}+a_t+0.8a_{t-1}$.}
$\Phi(B)=1-0.8B$ : racine $1.25>1$ $\Rightarrow$ \bonne{stationnaire}. $\Theta(B)=1+0.8B$ :
racine $-1.25$, module $>1$ $\Rightarrow$ \bonne{inversible}.

\textbf{Q\,B.7 --- Nature de quatre processus.}
\begin{enumerate}[label=\arabic*),leftmargin=*,nosep]
\item $X_t=\eps_t-\eps_{t-3}$ : MA(3) ($\theta_1=\theta_2=0,\theta_3=-1$). $\E=0$
(\bonne{centré}), somme finie de bruits $\Rightarrow$ \bonne{stationnaire} ;
\bonne{type ARMA/MA}, ARIMA$(0,0,3)$ (non intégré).
\item $X_t=\sigma^2-\eps_t^2$ : $\E(X_t)=\sigma^2-\E(\eps_t^2)=0$ (\bonne{centré}). Les
$\eps_t^2$ i.i.d.\ $\Rightarrow X_t$ i.i.d., $\gamma(k)=0$ ($k\ne0$) : \bonne{bruit blanc},
donc \bonne{stationnaire}, ARMA$(0,0)$.
\item $X_t-X_{t-2}=\eps_t$ : $(1-B^2)X_t=\eps_t$, $(1-B^2)=(1-B)(1+B)$ : racines $\pm1$ sur le
cercle $\Rightarrow$ \bonne{non stationnaire, intégré}.
\item $X_t=2t+\eps_t-\eps_{t-2}$ : $\E(X_t)=2t$ dépend de $t$ $\Rightarrow$ \bonne{non centré,
non stationnaire} ; processus \emph{TS} ($X_t-2t$ est un MA(2)), pas un ARIMA pur.
\end{enumerate}

\textbf{Q\,B.8 --- $X_t-aX_{t-1}=\eps_t$, $0<|a|<1$.}
\emph{1)} AR(1). Racine de $1-aB$ : $B=1/a$, $|1/a|>1$ $\Rightarrow$ \bonne{causal,
stationnaire}.
\emph{2)} $X_t=(1-aB)^{-1}\eps_t=\sum_{j\ge0}a^j\eps_{t-j}$.
\emph{3)} Si $(\eps_t)$ i.i.d.\ gaussien, $X_t$ (combinaison linéaire de gaussiennes) est un
\bonne{processus gaussien} (strictement stationnaire).

\textbf{Q\,B.9 --- $X_t-aX_{t-1}=\eps_t-2b\eps_{t-1}+b^2\eps_{t-2}$.}
$\Theta(B)=1-2bB+b^2B^2=(1-bB)^2$.
\emph{i)} ARMA(1,2) ; causal ssi $|a|<1$.
\emph{ii)} Inversible ssi racine de $(1-bB)^2=0$, $B=1/b$, de module $>1$, soit $|b|<1$ ; alors
$\eps_t=(1-bB)^{-2}(1-aB)X_t=\Big(\sum_{j\ge0}(j+1)b^jB^j\Big)(1-aB)X_t$.

\textbf{Q\,B.10 --- Différences saisonnières.}
\emph{1)} $(1-B^{12})s_t=0$ (car $s_t=s_{t-12}$) et $(1-B^{12})(\alpha+\beta t)=12\beta$
(constante), puis $(1-B)\cdot12\beta=0$. Donc
$(1-B)(1-B^{12})Z_t=(1-B)(1-B^{12})X_t$ : combinaison finie de $X$ stationnaire $\Rightarrow$
stationnaire.
\emph{2)} $(1-B^{12})[(\alpha+\beta t)s_t]=12\beta s_t$ (car $s_t$ de période 12), puis une
seconde application $(1-B^{12})[12\beta s_t]=0$. Donc $(1-B^{12})^2Z_t=(1-B^{12})^2X_t$,
stationnaire.

\textbf{Q\,B.11 --- $Y_t=X_{2t}$.}
\begin{etapes}
$X_{2t}=\alpha X_{2t-1}+a_{2t}=\alpha(\alpha X_{2t-2}+a_{2t-1})+a_{2t}
=\alpha^2X_{2t-2}+\alpha a_{2t-1}+a_{2t}$, soit $Y_t=\alpha^2Y_{t-1}+b_t$ avec
$b_t=\alpha a_{2t-1}+a_{2t}$.
\emph{Blancheur de $b_t$} : $\E(b_t)=0$ ; $\Var(b_t)=(\alpha^2+1)\sigma^2$ ;
$\Cov(b_t,b_{t-1})=\Cov(\alpha a_{2t-1}+a_{2t},\alpha a_{2t-3}+a_{2t-2})=0$ (indices distincts).
\end{etapes}
\begin{reponse}$(Y_t)$ est un \bonne{AR(1) de paramètre $\phi=\alpha^2$}, $|\alpha^2|<1$ :
stationnaire et causal.\end{reponse}

\textbf{Q\,B.12 --- $(1-\phi B^2)Y_t=\delta+(1+\theta B)a_t$.}
\emph{1)} $Y_t=\frac{\delta}{1-\phi}+(\sum_{i\ge0}\phi^iB^{2i})(1+\theta B)a_t$ ; d'où
$\psi_{2i}=\phi^i$, $\psi_{2i+1}=\theta\phi^i$ (donc $\psi_0=1,\psi_1=\theta,\psi_2=\phi,\dots$).
\emph{2)} Stationnaire/causal ssi racines de $1-\phi B^2$ de module $>1$, soit $|\phi|<1$
(inversibilité : $|\theta|<1$).
\emph{3)} $\sum\psi_j^2=\frac{1+\theta^2}{1-\phi^2}$ et
$\sum\psi_j\psi_{j+1}=\frac{\theta(1+\phi)}{1-\phi^2}=\frac{\theta}{1-\phi}$, d'où
$\rho(1)=\dfrac{\theta(1+\phi)}{1+\theta^2}$.

\section{Représentations \texorpdfstring{$\mathrm{MA}(\infty)$ et $\mathrm{AR}(\infty)$}{MA-inf et AR-inf}}
\markboth{Représentations MA-infini et AR-infini}{}

\textbf{Q\,C.1 --- AR(2) $X_t-0.9X_{t-1}+0.2X_{t-2}=a_t$, coefficients $\psi_k$.}
\begin{etapes}
$\Phi(B)\Psi(B)=1$ avec $\Phi=1-0.9B+0.2B^2$. En identifiant $B^k$ :
$\psi_k-0.9\psi_{k-1}+0.2\psi_{k-2}=0$, soit
$\psi_k=0.9\psi_{k-1}-0.2\psi_{k-2}$ ($\psi_0=1,\psi_{-1}=0$) :
\[
\psi_1=0.9,\ \psi_2=0.9(0.9)-0.2(1)=0.61,\ \psi_3=0.9(0.61)-0.2(0.9)=0.369.
\]
\emph{Forme générale.} Racines de $0.2B^2-0.9B+1=0$ : $B=2.5$ et $B=2$, d'inverses $0.4$ et
$0.5$. Donc $\psi_k=A(0.5)^k+C(0.4)^k$. Avec $\psi_0=A+C=1$ et $\psi_1=0.5A+0.4C=0.9$ :
$A=5,\ C=-4$.
\end{etapes}
\begin{reponse}$\psi_1=0.9,\ \psi_2=0.61,\ \psi_3=0.369$ ;
$\boxed{\psi_k=5(0.5)^k-4(0.4)^k}$ (vérif.\ $\psi_2=1.25-0.64=0.61$).\end{reponse}

\textbf{Q\,C.2 --- $X_t=0.6X_{t-1}+0.2X_{t-2}+a_t-0.5a_{t-1}$, coefficients $\pi_k$.}
$\Phi(B)=1-0.6B-0.2B^2$ (donc $\phi^\Phi_1=-0.6,\ \phi^\Phi_2=-0.2$), $\Theta(B)=1-0.5B$
($\theta^\Theta_1=-0.5$).
\begin{etapes}
$\Theta(B)\Pi(B)=\Phi(B)$ donne, pour le coefficient de $B^k$ : $\pi_k-0.5\pi_{k-1}=\phi^\Phi_k$.
\begin{align*}
k=1:&\ \pi_1-0.5(1)=-0.6\Rightarrow \pi_1=-0.1.\\
k=2:&\ \pi_2-0.5(-0.1)=-0.2\Rightarrow \pi_2=-0.2-0.05=-0.25.\\
k=3:&\ \pi_3-0.5(-0.25)=0\Rightarrow \pi_3=-0.125.
\end{align*}
\end{etapes}
\begin{reponse}$\pi_0=1,\ \pi_1=-0.1,\ \pi_2=-0.25,\ \pi_3=-0.125$ ; soit
$a_t=X_t-0.1X_{t-1}-0.25X_{t-2}-0.125X_{t-3}-\dots$\end{reponse}

\textbf{Q\,C.3 --- $(1-0.8B)X_t=(1+0.4B)a_t$.}
\begin{etapes}
\emph{AR$(\infty)$} : $\Theta(B)\Pi(B)=\Phi(B)$, $\Theta=1+0.4B$, $\Phi=1-0.8B$ :
$\pi_k+0.4\pi_{k-1}=\phi^\Phi_k$ ($\phi^\Phi_1=-0.8$, sinon $0$).
$\pi_1=-0.8-0.4=-1.2$ ; $\pi_2=-0.4(-1.2)=0.48$ ; $\pi_3=-0.4(0.48)=-0.192$.
\emph{MA$(\infty)$} : $\Phi(B)\Psi(B)=\Theta(B)$ : $\psi_k-0.8\psi_{k-1}=\theta^\Theta_k$
($\theta^\Theta_1=0.4$, sinon $0$).
$\psi_1=0.4+0.8=1.2$ ; $\psi_2=0.8(1.2)=0.96$ ; $\psi_3=0.8(0.96)=0.768$.
\end{etapes}
\begin{reponse}AR : $\pi_1=-1.2,\ \pi_2=0.48,\ \pi_3=-0.192$ ;\quad
MA : $\psi_1=1.2,\ \psi_2=0.96,\ \psi_3=0.768$ (et $\psi_k=1.2(0.8)^{k-1}$).\end{reponse}

\textbf{Q\,C.4 --- $(1-\Phi_1B^7)x_t=a_t$.}
\emph{a)} $x_t=\sum_{j\ge0}\Phi_1^{\,j}a_{t-7j}$ ($\psi_{7j}=\Phi_1^j$, autres nuls).
\emph{b)} $\gamma_0=\sigma^2\sum_{j\ge0}\Phi_1^{2j}=\dfrac{\sigma^2}{1-\Phi_1^2}$.
\emph{c)} Yule--Walker saisonnier : $\rho(k)=\Phi_1\rho(k-7)$, nul si $7\nmid k$. D'où
$\rho_7=\Phi_1$, $\rho_{14}=\Phi_1^2$, $\rho_1=\rho_8=0$.
\emph{d)} FAP : $P_7=\Phi_1$, puis $P_{14}=0$ ; $P_1=P_8=0$.

\textbf{Q\,C.5 --- AR(2) $X_t-0.9X_{t-1}+0.2X_{t-2}=a_t$.}
\emph{1)} Racines $\{2,2.5\}$, module $>1$ : \bonne{stationnaire}.
\emph{2)} $\rho_1=\frac{\phi_1}{1-\phi_2}=\frac{0.9}{1.2}=0.75$ ;
$\rho_2=\phi_1\rho_1+\phi_2=0.9(0.75)-0.2=0.475$.
\emph{3)} $\rho_k=0.9\rho_{k-1}-0.2\rho_{k-2}$ ; forme $\rho_k=3.5(0.5)^k-2.5(0.4)^k$
(constantes via $\rho_0=1,\rho_1=0.75$).
\emph{4)} $\psi_k=5(0.5)^k-4(0.4)^k$ (Q\,C.1).

\textbf{Q\,C.6 --- Identifier $\phi,\theta$ depuis $\psi_k$.}
Pour $x_t=\phi x_{t-1}+a_t-\theta a_{t-1}$ : $\psi_0=1$, $\psi_1=\phi-\theta$, et
$\psi_k=\phi\psi_{k-1}$ ($k\ge2$). La suite donnée est $1,\ 0.3,\ 0.15,\ 0.075,\dots$
\begin{etapes}
Rapport stabilisé : $\dfrac{\psi_2}{\psi_1}=\dfrac{0.15}{0.3}=0.5=\phi$
(et $\psi_3/\psi_2=0.075/0.15=0.5$ : cohérent).
Puis $\psi_1=\phi-\theta=0.3\Rightarrow\theta=\phi-0.3=0.5-0.3=0.2$.
\end{etapes}
\begin{reponse}$\boxed{\phi=0.5,\quad \theta=0.2}$.\end{reponse}

\section{Autocorrélations ACF et PACF}
\markboth{Autocorrélations ACF / PACF}{}

\textbf{Q\,D.1 --- AR(2), $\phi_1=0.85,\phi_2=-0.32$.}
\emph{1)} AR(2) : la FAP coupe à 2, $P_1=\rho_1$, $P_2=\phi_2$. D'où
\begin{reponse}$\phi_{11}=\rho(1)=0.84$ et $\phi_{22}=\phi_2=-0.32$.\end{reponse}
\emph{2)} $\rho(1)=\frac{\phi_1}{1-\phi_2}$, $\rho(2)=\phi_1\rho(1)+\phi_2$.
\emph{3)} $\rho(1)=\frac{0.85}{1.32}=0.6439$ ; $\rho(2)=0.85(0.6439)-0.32=0.2273$.

\textbf{Q\,D.2 --- $(1-0.8B^2)X_t=a_t$.}
AR(2), $\phi_1=0,\phi_2=0.8$. $\rho_1=\frac{0}{1-0.8}=0$ ; $\rho_2=\phi_1\rho_1+\phi_2=0.8$.
$P_1=\rho_1=0$ ; $P_2=\frac{\rho_2-\rho_1^2}{1-\rho_1^2}=0.8=\phi_2$.
\begin{reponse}$\rho_1=0,\ \rho_2=0.8$ ; $P_1=0,\ P_2=0.8$.\end{reponse}

\textbf{Q\,D.3 --- $X_t=0.6X_{t-1}+0.8X_{t-2}-0.3X_{t-3}+a_t$.}
\begin{attention}
$\Phi(B)=1-0.6B-0.8B^2+0.3B^3$. Les racines ont pour modules $2.96,\ 1.22,\ 0.92$ : l'une
est \emph{$<1$}, donc \textbf{le processus n'est pas stationnaire} et l'ACF théorique n'est pas
définie au sens usuel.
\end{attention}
Si l'on suppose néanmoins la récurrence valide, Yule--Walker donne
$\rho_k=0.6\rho_{k-1}+0.8\rho_{k-2}-0.3\rho_{k-3}$ avec le système des deux premières équations
$\begin{cases}0.2\rho_1+0.3\rho_2=0.6\\0.3\rho_1-\rho_2=-0.8\end{cases}$, puis
$\rho_3=0.6\rho_2+0.8\rho_1-0.3$. \emph{(La conclusion attendue est la non-stationnarité.)}

\textbf{Q\,D.4 --- AR(2) $X_t-0.9X_{t-1}+0.2X_{t-2}=a_t$.}
Identique à Q\,C.5 : $\rho_1=0.75$, $\rho_2=0.475$, $\rho_k=0.9\rho_{k-1}-0.2\rho_{k-2}
=3.5(0.5)^k-2.5(0.4)^k$ ($k\ge3$).

\textbf{Q\,D.5 --- Identification par corrélogrammes.}
ACF : $0.65,0.47,0.35,0.24,0.18$ \emph{décroît} régulièrement (pas de coupure). PACF :
$0.65$ puis $|P_k|\lesssim0.2=2/\sqrt{100}$ (non significatifs) : \emph{coupure après le rang 1}.
\begin{reponse}\bonne{AR(1)} : $\rho_k\approx0.65^k$ ($0.65,0.42,0.27,\dots$, proche de
l'observé) et PACF qui coupe à 1.\end{reponse}

\textbf{Q\,D.6 --- $z_t=x_t+y_t$, AR(1)+MA(1) indépendants.}
La somme d'un ARMA$(1,0)$ et d'un ARMA$(0,1)$ indépendants est un ARMA$(1,\max(1+1,0+0))=$ARMA$(1,2)$
\emph{apparent}, qui se réduit (calcul des autocovariances) à un \bonne{ARMA$(1,1)$}.

\section{Identification graphique}
\markboth{Identification graphique}{}

\textbf{Q\,I.1 --- Associer série et graphique.}
\begin{reponse}
(1) $\eps_t$ $\to$ panneau bruit pur. (2) $s_t+\eps_t$ $\to$ saison centrée sur 0.
(3) $a+s_t+\eps_t$ $\to$ saison autour d'un niveau $>0$. (4) $a\,s_t\,\eps_t$ $\to$ amplitude
\emph{croissante} (multiplicatif). (5) $f_t+\eps_t$ $\to$ ondulation lente $+$ bruit additif.
(6) $f_t\,\eps_t$ $\to$ bruit dont l'amplitude suit $f_t$ (multiplicatif).
\end{reponse}
Critères : présence d'un \emph{niveau} ($a$), d'une \emph{saisonnalité} rapide ($s$) ou d'une
\emph{ondulation lente} ($f$), et caractère additif (amplitude constante) ou multiplicatif
(amplitude proportionnelle au niveau).

\textbf{Q\,I.2 --- Identifier D, E, F par l'ACF.}
\begin{reponse}
\textbf{Série D} : ACF qui \emph{décroît géométriquement} $\to$ \bonne{AR(1)}.
\textbf{Série E} : ACF qui \emph{s'annule après quelques retards} $\to$ \bonne{MA} (ordre $=$
dernier retard significatif). \textbf{Série F} : ACF qui \emph{décroît très lentement}
($\approx1$) $\to$ \bonne{processus non stationnaire / intégré}.
\end{reponse}

\textbf{Q\,I.3 --- ACF/PACF (17/12/2019).}
\emph{1)} PACF coupe à 1, ACF décroît $\to$ AR(1) ; (ACF coupe à 1 $\to$ MA(1), etc.).
\emph{2)} ACF qui décroît \emph{très lentement} $\to$ ARIMA$(1,1,0)$.
\emph{3)} PACF coupe à 2 $\to$ AR(2) (à 3 $\to$ AR(3)).
\emph{4)--5)} Pics aux retards saisonniers (multiples de $s$) $\to$ partie SARIMA ; ex.\ un seul
pic saisonnier $\to$ $(1,0,0)\times(1,0,0)_{12}$.
