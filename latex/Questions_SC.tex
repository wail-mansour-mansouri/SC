\section{QCM théoriques et questions de cours}
\markboth{QCM théoriques}{}

\subsection{Concepts de base et décomposition}

\begin{enumerate}[label=\textbf{Q\,A.\arabic*},leftmargin=*]

\item Un processus \textbf{stationnaire} :
\begin{enumerate}[label=\Alph*.]
  \item permet de modéliser des séries temporelles qui ne peuvent pas présenter une
        tendance ou une saisonnalité ;
  \item permet de modéliser des séries présentant une tendance ;
  \item permet de modéliser des séries présentant une saisonnalité ;
  \item permet de modéliser des séries présentant une tendance et une saisonnalité.
\end{enumerate}

\item Une \textbf{moyenne mobile} est :
\begin{enumerate}[label=\Alph*.]
  \item un filtre non linéaire ;
  \item un filtre linéaire, pas forcément centré ou symétrique ;
  \item un filtre linéaire centré, pas forcément symétrique ;
  \item un filtre linéaire normalisé.
\end{enumerate}

\item La \textbf{moyenne mobile MM4} :
\begin{enumerate}[label=\Alph*.]
  \item absorbe les tendances linéaires et conserve les saisonnalités de période 4 ;
  \item absorbe les tendances linéaires et conserve les saisonnalités de période paire ;
  \item conserve les tendances linéaires et absorbe les saisonnalités de période 4 ;
  \item conserve les tendances linéaires et absorbe les saisonnalités de période impaire.
\end{enumerate}

\item L'algorithme \textbf{X11} :
\begin{enumerate}[label=\Alph*.]
  \item contient deux boucles pour évaluer la série CVS ;
  \item est l'algorithme le plus sophistiqué connu à l'heure actuelle ;
  \item intègre de manière conjointe régression linéaire et moyenne mobile.
\end{enumerate}

\item \textbf{Vrai / Faux.} La décomposition d'une série chronologique est essentiellement
utilisée pour retrancher à une série sa saisonnalité.

\item \textbf{Vrai / Faux.} Idéalement, la première moyenne mobile utilisée dans
l'extraction des variations saisonnières d'une série doit conserver la tendance et réduire
la variance du bruit.

\item \textbf{Vrai / Faux.} La prévision obtenue par lissage exponentiel simple ne tient
pas compte de la saisonnalité.

\item La \textbf{méthode de Holt--Winters} \emph{(plusieurs réponses possibles)} :
\begin{enumerate}[label=\Alph*.]
  \item est plus efficace que les lissages exponentiels simple ou double ;
  \item tient compte d'une éventuelle saisonnalité mais pas de tendance ;
  \item tient compte d'une éventuelle tendance et saisonnalité.
\end{enumerate}
\end{enumerate}

\subsection{Caractérisation des processus AR, MA, ARMA, SARIMA}

\begin{enumerate}[label=\textbf{Q\,A.\arabic*},leftmargin=*,start=9]

\item Un processus \textbf{AR} est caractérisé par :
\begin{enumerate}[label=\Alph*.]
  \item des autocorrélations simples et partielles s'annulant à partir d'un certain rang ;
  \item des autocorrélations simples décroissant rapidement vers 0 et des autocorrélations
        partielles s'annulant à partir d'un certain rang ;
  \item des autocorrélations simples s'annulant à partir d'un certain rang et des
        autocorrélations partielles décroissant rapidement vers 0.
\end{enumerate}

\item Un processus \textbf{ARMA} est caractérisé par :
\begin{enumerate}[label=\Alph*.]
  \item des autocorrélations simples et partielles s'annulant à partir d'un certain rang ;
  \item des autocorrélations simples s'annulant à partir d'un certain rang ;
  \item des autocorrélations partielles s'annulant à partir d'un certain rang ;
  \item aucune des propriétés précédentes.
\end{enumerate}

\item Un processus \textbf{SARIMA} :
\begin{enumerate}[label=\Alph*.]
  \item est stationnaire ;
  \item permet de modéliser des séries présentant une tendance mais pas de saisonnalité ;
  \item permet de modéliser des séries présentant une saisonnalité mais pas de tendance ;
  \item permet de modéliser des séries présentant une tendance et/ou une saisonnalité.
\end{enumerate}

\item \textbf{Vrai / Faux.} Tout processus MA stationnaire possède une représentation AR.

\item Afin de \textbf{valider un modèle ARMA}, il faut vérifier
\emph{(plusieurs réponses possibles)} :
\begin{enumerate}[label=\Alph*.]
  \item la significativité des paramètres ;
  \item la blancheur du résidu ;
  \item la normalité des résidus (ou de la série ajustée).
\end{enumerate}

\item \textbf{Vrai / Faux.} Si $X_t$ est stationnaire, alors on espère que la statistique de
Ljung--Box calculée sur $X_t$ devrait nous amener à accepter l'hypothèse nulle qui lui est
associée.
\end{enumerate}

\section{Stationnarité, causalité et inversibilité}
\markboth{Stationnarité, causalité, inversibilité}{}

\begin{enumerate}[label=\textbf{Q\,B.\arabic*},leftmargin=*]

\item Soit une variable aléatoire $z$ d'espérance nulle et de variance unitaire à partir de
laquelle on construit la série $X_t=(-1)^t z$. Étudier l'espérance et les autocovariances de
$X_t$. Le processus est-il stationnaire au sens large ?

\item Soit l'équation $X_t=0.8X_{t-1}-0.15X_{t-2}+a_t-0.3a_{t-1}$, où $a_t$ est un bruit
blanc. Le processus $X_t$ est-il :
\emph{(a)} stationnaire et inversible ; \emph{(b)} stationnaire et non inversible ;
\emph{(c)} non stationnaire et inversible ; \emph{(d)} non stationnaire et non inversible ?

\item \textbf{Vrai / Faux.} Le processus $(1-1.66B+0.66B^2)X_t=(1-B)a_t$, où $a_t$ est un
bruit blanc, possède une représentation AR d'ordre fini. Justifier.

\item \textbf{Vrai / Faux.} Si $X_t=1.6X_{t-1}-0.6X_{t-2}+a_t-0.6a_{t-1}$ (avec $a_t$ bruit
blanc), alors $X_t$ est stationnaire. Justifier.

\item Soit le processus $(1-0.8B)X_t=(1+0.4B)a_t$.
\begin{enumerate}[label=\alph*.]
  \item Ce processus possède-t-il une représentation AR ? (non / oui, ordre fini / oui,
        ordre infini). Justifier.
  \item Ce processus possède-t-il une représentation MA ? (non / oui, ordre fini / oui,
        ordre infini). Justifier.
\end{enumerate}

\item Le processus $X_t=0.8X_{t-1}+a_t+0.8a_{t-1}$ est-il stationnaire ?

\item Pour chacun des processus suivants ($(\eps_t)$ bruit blanc de variance $\sigma^2>0$),
dire en justifiant s'il est centré et/ou stationnaire et/ou de type ARIMA :
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item $X_t=\eps_t-\eps_{t-3}$ ;
  \item $X_t+\eps_t^2=\sigma^2$ ;
  \item $X_t-X_{t-2}=\eps_t$ ;
  \item $X_t=2t+\eps_t-\eps_{t-2}$.
\end{enumerate}

\item Soit $(X_t)$ vérifiant $X_t-aX_{t-1}=\eps_t$, $t\in\Z$, avec $(\eps_t)$ bruit blanc de
variance $\sigma^2>0$ et $0<|a|<1$.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Reconnaître le processus. Est-il causal ? Justifier.
  \item Donner l'expression de $X_t$ en fonction du bruit blanc $(\eps_t)$.
  \item Si les $(\eps_t)$ sont i.i.d. de loi normale, que peut-on dire de $(X_t)$ ?
\end{enumerate}

\item Soit $(X_t)$ vérifiant $X_t-aX_{t-1}=\eps_t-2b\eps_{t-1}+b^2\eps_{t-2}$, $t\in\Z$, avec
$(\eps_t)$ bruit blanc de variance $\sigma^2>0$, $a,b\in\R$.
\begin{enumerate}[label=\roman*),leftmargin=*]
  \item Reconnaître le processus. Est-il causal ? Justifier.
  \item Quand $(X_t)$ est-il inversible ? Exprimer alors $(\eps_t)$ en fonction de $(X_t)$.
\end{enumerate}

\item \emph{(Examen 22/12/2011.)} Soit $X_t$ stationnaire au sens large, de moyenne nulle,
et $\alpha,\beta\in\R$.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Si $Z_t=\alpha+\beta t+s_t+X_t$ où $s_t$ est une saisonnalité de période 12,
        démontrer que $(1-B)(1-B^{12})Z_t$ est stationnaire.
  \item Si $Z_t=(\alpha+\beta t)s_t+X_t$, démontrer que $(1-B^{12})^2Z_t$ est stationnaire.
\end{enumerate}

\item \emph{(Examen 22/12/2011.)} Soit l'AR(1) $X_t=\alpha X_{t-1}+a_t$, $a_t\sim\BB(0,\sigma^2)$,
$|\alpha|<1$. On pose $Y_t=X_{2t}$. Montrer que $Y_t=\phi Y_{t-1}+b_t$ avec $b_t$ bruit blanc
et $\phi$ à identifier. En déduire que $Y_t$ est un AR(1) stationnaire et causal.

\item Soit l'équation $(1-\phi B^2)Y_t=\delta+(1+\theta B)a_t$.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Déterminer la suite $(\psi_j)$ telle que $Y_t=\sum_{j\ge0}\psi_j a_{t-j}$.
  \item Donner les conditions sur $\theta$ et $\phi$ pour que $Y_t$ soit stationnaire au sens
        large et causal.
  \item Calculer $\rho(1)$.
\end{enumerate}
\end{enumerate}

\section{Intégration, ARIMA et tests de Dickey--Fuller}

\item Soit $x_t=2.7x_{t-1}-2.4x_{t-2}+0.7x_{t-3}+a_t$, $a_t$ bruit blanc.

\item Le processus $X_t=1.6X_{t-1}-0.3X_{t-2}+0.6X_{t-3}+a_t$ est-il intégré ?

\item \emph{(Série E, AR(1) et test de Dickey--Fuller.)} On dispose, pour la série $Y_t$
($t=2,\dots,300$), de : $\bar y=2.998$, $\sum Y_tY_{t-1}=25\,842.3$,

\begin{enumerate}[label=\alph*.]
  \item Estimer $\phihat$ du modèle AR(1) $Y_t=\phi Y_{t-1}+a_t$ par MCO.
  \item Calculer l'écart-type estimé de $\phihat$ sachant $\hat\sigma_a^2=0.954$.
  \item Effectuer le test de Dickey--Fuller simple $H_0:\phi=1$ contre $H_1:\phi<1$
        (valeur critique à 5\% : $-2.89$). Conclure.
  \item Après différenciation $\Delta Y_t$, on obtient $s_{\Delta Y}^2=0.954$ et
        $\rhohat(1)=-0.035$. Que conclure sur la nature du processus et le modèle approprié ?
  \item Comparer les variances avant/après différenciation. La différenciation réduit-elle
        toujours la variance ?
\end{enumerate}

\item On a simulé 300 observations de $Y$ (figure 1). Le générateur est l'un des suivants
($a_t$ bruit blanc gaussien $\sigma=1$). Lequel a vraisemblablement été employé ?
Justifier et expliquer pourquoi on écarte les autres.
\begin{enumerate}[label=\alph*),leftmargin=*]
  \item $Y_t=1.8Y_{t-1}+a_t$ ;

\section{Exercices de synthèse}
\markboth{Exercices de synthèse}{}

\begin{enumerate}[label=\textbf{Q\,J.\arabic*},leftmargin=*]

\item Soit $X_t=\phi X_{t-1}+\eps_t$ un AR(1), $(\eps_t)$ bruit blanc, $\phi\ne0$, $(X_t)$
stationnaire.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Laquelle de ces affirmations n'est \emph{pas} correcte :
  (A) $\eps_t$ et $X_{t-1}$ non corrélés ; (B) $X_t$ et $\eps_{t-1}$ non corrélés ;
  (C) $\eps_t$ et $\eps_{t-1}$ non corrélés ; (D) $X_t$ et $\eps_t$ corrélés.
  \item Laquelle est correcte : (A) $\cor(X_t,X_{t-1})=\phi$ ; (B) $\cor(X_t,X_{t-2})=0$ ;
  (C) $\cor(X_t,\eps_t)=0$ ; (D) $\cor(X_t,\eps_t)=\phi$.
  \item On suppose $\sigma_\eps^2=1$. Que vaut $\Var(X_t)$ :
  (A) 1 ; (B) $1-\phi^2$ ; (C) $\dfrac{1}{1-\phi^2}$ ; (D) 0 ?
  \item Une seule affirmation est correcte : (A) $Y_t=t+X_t$ est stationnaire ;
  (B) $Y_t=X_t-X_{t-1}$ est stationnaire ; (C) $Y_t=X_t+Y_{t-1}$ est stationnaire ;
  (D) $Y_t=\eps_t+\phi^{-1}Y_{t-1}$ est stationnaire.
\end{enumerate}

\item AR(1) sur données journalières : on observe $-1$ hier et $+1$ aujourd'hui.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Quelle prévision pour demain ? (A) $\phi$ ; (B) $-\phi$ ; (C) 0 ; (D) $\phi+\eps$.
  \item Quelle est la variance de l'erreur de prévision à deux jours ?
        (A) $\phi$ ; (B) 1 ; (C) $1+\phi^2$ ; (D) $\phi^2$.
\end{enumerate}

\item ARMA(1,1) : $X_t=\phi X_{t-1}+\eta_t+\theta\eta_{t-1}$, $(\eta_t)$ bruit blanc de
variance $\sigma^2$.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Condition nécessaire et suffisante de stationnarité.
  \item Parmi plusieurs affirmations, laquelle est toujours vraie (cf.\ propriété
        $\rho(h)=\phi\rho(h-1)$ pour $h$ assez grand) ?
  \item Si $\phi+\theta=0$, que devient $(X_t)$ ? (marche aléatoire / bruit blanc / AR(1) /
        MA(1)).
\end{enumerate}

\item AR$(p)$ stationnaire, $X_t=\phi_1X_{t-1}+\dots+\phi_pX_{t-p}+u_t$, $\phi_p\ne0$.
Une seule affirmation est (en général) vraie : (A) $\rho(h)=0$ pour $h>p$ ;
(B) $P(h)=0$ pour $h>p$ ; (C) $\rho(p)\ne0$ ; (D) $\rho(p)=1$.

\item $Y_t=Y_{t-1}+\eps_t$ (marche aléatoire), $\sigma_\eps^2=0.18$, $Y_t=0$ pour $t\le0$.
Quel est l'écart-type de $Y_2$ ?

\item $Y_t=\dfrac{\eps_t+\eps_{t-1}}{2}$, $\sigma_\eps^2=8$. Calculer $\Cov(Y_t,Y_{t-1})$.

\item $Y_t=\dfrac{\eps_t+\eps_{t-1}+\dots+\eps_{t-7}}{8}$, $\sigma_\eps^2=0.64$.
Calculer $\Corr(Y_t,Y_{t-1})$.

\item AR(1) stationnaire, 200 observations, $\gammahat(0)=0.348$, $\Var(\bar Y)=0.006$.
Quelle est la valeur de $\phi$ ?

\item MA(2) avec $\theta_1=0.2$, $\theta_2=-0.7$, $\sigma_\eps^2=1$.
Calculer $\Var(Y_t)$, $\Corr(Y_t,Y_{t-1})$ et $\Corr(Y_t,Y_{t-2})$.

\item AR(1) avec $\phi=-0.5$ et $\sigma_\eps=8.66$. Calculer $\Cov(Y_t,Y_{t-1})$ et
$\Cov(Y_t,Y_{t-2})$.

\item AR(1) avec $\phi>0$, $\sigma_\eps^2=5.333$ et $\gamma(2)=3$.
Déterminer $\phi$ et $\gamma(1)$.

\item ARMA(1,1) avec $\phi=-0.5$, $\theta=-0.6$, $\sigma_\eps^2=4$.
Calculer $\gamma(0),\gamma(1),\rho(1),\rho(2)$.

\item ARIMA(2,1,2) :
$Y_t=1+1.8Y_{t-1}-1.4Y_{t-2}+0.6Y_{t-3}+\eps_t-0.4\eps_{t-1}+0.1\eps_{t-2}$.
Déterminer $\phi_1$ et $\phi_2$ (de la partie ARMA après différenciation).

\item $Y_t=X_t+\eps_t$ avec $X_t=X_{t-1}+u_t$ ; $(\eps_t)$ et $(u_t)$ bruits blancs
indépendants, $\sigma_\eps^2=2$, $\sigma_u^2=1$.
Calculer $\Var(\Delta Y_t)$ et $\Corr(\Delta Y_t,\Delta Y_{t-1})$.

\item 100 observations d'un AR(2) : $\gammahat(0)=4$, $\rhohat(1)=0.5$, $\rhohat(2)=0.5$,

quelle est la valeur de $Y_4$ ?

$V_t,a_t$ deux bruits blancs gaussiens indépendants de variances $\sigma_v^2,\sigma_a^2$.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Donner la fonction d'autocovariance de $Y_t$ en fonction de $\phi,\sigma_a^2,\sigma_v^2$.
  \item Le processus $Y_t$ est-il stationnaire ?

\section{Hors programme du cours BADAOUI\,\texorpdfstring{\horsprog}{*}}
\markboth{Hors programme}{}
\begin{tcolorbox}[colback=orangelight,colframe=orange,title={Avertissement}]
Les questions ci-dessous portent sur des notions \textbf{non couvertes} (ou seulement
évoquées) par le cours de F.~BADAOUI. Elles proviennent d'autres sources (QCM Charpentier,
examens « INSEA~2016/2017 », « 17/12/2019 », OpenClassroom). Elles sont traitées à part dans
le corrigé.
\end{tcolorbox}

\begin{enumerate}[label=\textbf{Q\,Z.\arabic*},leftmargin=*]

\item \textbf{Bruit blanc fort / faible.} Soit $(\eta_t)$ i.i.d.\ centrée de variance unité.
Vérifier que $\eps_t=\eta_t\eta_{t-1}\eta_{t-2}$ est un bruit blanc, préciser s'il est fort ou
faible, puis montrer que $(\eps_t^2)$ suit un modèle ARMA.

\item \textbf{Stationnarité mutuelle.} Pour que deux processus $X$ et $Y$ soient mutuellement
stationnaires, il faut que : $X$ soit stationnaire ; $Y$ soit stationnaire ; la covariance
croisée de $X$ et $Y$ soit invariante par translation dans le temps ; \dots
(cocher les bonnes réponses).

\item \textbf{Inverses d'opérateurs.} Donner l'inverse des opérateurs
$I-\tfrac15 B$ et $I+3B$. Préciser le domaine de validité.

\item \textbf{Projection orthogonale (examen 17/12/2019, Ex.~3).}
Soit $\Phi(B)=I-\tfrac16 B-\tfrac16 B^2$ et $(\eps_t)$ un bruit blanc de variance
$\sigma^2>0$.
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Montrer que $\Phi(B)$ est inversible et calculer son inverse.
  \item Pour $\Phi(B)X_t=\eps_t$ : type du processus, stationnarité, ordres ; forme
        $\mathrm{MA}(\infty)$ ; espérance et forme de $\gamma$ ; loi de $(X_t,X_{t+3})$ dans
        le cas gaussien.
  \item Pour $\Phi(B)Y_t=\eps_t-\tfrac13\eps_{t-1}$ : type, stationnarité, ordres ;
        minimalité ; $(\eps_t)$ est-il le processus des innovations ? écriture
        $\mathrm{AR}(\infty)$ ; projection orthogonale de $Y_t$ sur le passé.
\end{enumerate}

\item \textbf{Lissage / filtres (QCM OpenClassroom \& 17/12/2019).}
\begin{enumerate}[label=\arabic*),leftmargin=*]
  \item Le lissage exponentiel simple consiste à ajuster localement une constante à la
        série. (Vrai / Faux)
  \item Le lissage de Holt--Winters permet de prendre en compte une composante saisonnière.
        (Vrai / Faux)
  \item Pour estimer une tendance polynomiale déterministe $m$ en présence d'une
        saisonnalité $s$, on applique un filtre qui : absorbe $m$ et $s$ / absorbe $s$ et
        laisse $m$ / absorbe $m$ et laisse $s$ / laisse $m$ et $s$.
  \item Dans Box--Jenkins, pour tester la blancheur des résidus on peut : calculer l'AIC /
        regarder ACF et PACF empiriques / utiliser un test de Ljung--Box / appliquer un
        filtre moyenne mobile.
\end{enumerate}
\end{enumerate}
